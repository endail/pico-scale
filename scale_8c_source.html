88 bool scale_get_values_timeout(
89  scale_t* const sc,
90  int32_t** const arr,
91  size_t* const len,
92  const uint timeout) {
93 
94  assert(sc != NULL);
95  assert(sc->_hx != NULL);
96  assert(arr != NULL);
97  assert(len != NULL);
98 
99  int32_t val; //temporary value from the HX711
100  int32_t* memblock; //ptr to the memblock
101  size_t elemCount; //number of elements in the block
102 
123  elemCount = hx711_get_rate_sps(hx711_rate_80);
124 
125  if((*arr = malloc(elemCount * sizeof(int32_t))) == NULL) {
126  return false;
127  }
128 
129  *len = 0;
130 
131  while(hx711_get_value_timeout(sc->_hx, timeout, &val)) {
132 
133  //new value available, so increase the counter
134  //this is the actual number of values obtained
135  ++(*len);
136 
137  //check if a reallocation is needed
138  if(*len > elemCount) {
139 
140  //when a reallocation is needed, double the space
141  elemCount = elemCount * 2;
142 
143  memblock = realloc(
144  *arr,
145  elemCount * sizeof(int32_t));
146 
147  //if memory allocation fails, return false
148  //existing *arr will still be allocated
149  //but will be freed in caller function
150  if(memblock == NULL) {
151  return false;
152  }
153 
154  //move pointer of new block to *arr
155  *arr = memblock;
156 
157  }
158 
159  //store the value in the array
160  (*arr)[(*len) - 1] = val;
161 
162  }
163 
164  //no errors occurred, so return true
165  return true;
166

169 bool scale_read(
170  scale_t* const sc,
171  double* const val,
172  const scale_options_t* const opt) {
173 
174  assert(sc != NULL);
175  assert(val != NULL);
176  assert(opt != NULL);
177 
178  int32_t* arr = NULL;
179  size_t len = 0;
180  bool ok = false; //assume error
181 
182  switch(opt->strat) {
183  case strategy_type_time:
184  ok = scale_get_values_timeout(
185  sc,
186  &arr,
187  &len,
188  opt->timeout);
189  break;
190 
191  case strategy_type_samples:
192  default:
193  len = opt->samples;
194  ok = scale_get_values_samples(
195  sc,
196  &arr,
197  len);
198  break;
199  }
200 
201  //if an error occurred or no samples obtained
202  //deallocate any memory and then return false
203  if(!ok || len == 0) {
204  free(arr);
205  return false;
206  }
207 
208  switch(opt->read) {
209  case read_type_average:
210  util_average(arr, len, val);
211  break;
212 
213  case read_type_median:
214  default:
215  util_median(arr, len, val);
216  break;
217  }
218 
219  //at this point, statistical func is done
220  //so deallocate and return true
221  free(arr);
222  return true;
223 
224 }

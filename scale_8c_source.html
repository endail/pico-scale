88 bool scale_get_values_timeout(
89  scale_t* const sc,
90  int32_t** const arr,
91  size_t* const len,
92  const absolute_time_t* const timeout) {
93 
94  assert(sc != NULL);
95  assert(sc->_hx != NULL);
96  assert(arr != NULL);
97  assert(len != NULL);
98  assert(timeout != NULL);
99  assert(!is_nil_time(*timeout));
100 
101  int32_t val; //temporary value from the HX711
102  int32_t* memblock; //ptr to the memblock
103  size_t elemCount; //number of elements in the block
104 
125  elemCount = hx711_get_rate_sps(hx711_rate_80);
126 
127  if((*arr = malloc(elemCount * sizeof(int32_t))) == NULL) {
128  return false;
129  }
130 
131  *len = 0;
132 
133  while(hx711_get_value_timeout(sc->_hx, timeout, &val)) {
134 
135  //new value available, so increase the counter
136  //this is the actual number of values obtained
137  ++(*len);
138 
139  //check if a reallocation is needed
140  if(*len > elemCount) {
141 
142  //when a reallocation is needed, double the space
143  elemCount = elemCount * 2;
144 
145  memblock = realloc(
146  *arr,
147  elemCount * sizeof(int32_t));
148 
149  //if memory allocation fails, return false
150  //existing *arr will still be allocated
151  //but will be freed in caller function
152  if(memblock == NULL) {
153  return false;
154  }
155 
156  //move pointer of new block to *arr
157  *arr = memblock;
158 
159  }
160 
161  //store the value in the array
162  (*arr)[(*len) - 1] = val;
163 
164  }
165 
166  //no errors occurred, so return true
167  return true;
168 
169 }

171 bool scale_read(
172  scale_t* const sc,
173  double* const val,
174  const scale_options_t* const opt) {
175 
176  assert(sc != NULL);
177  assert(val != NULL);
178  assert(opt != NULL);
179 
180  int32_t* arr = NULL;
181  size_t len = 0;
182  absolute_time_t timeout;
183  bool ok = false; //assume error
184 
185  switch(opt->strat) {
186  case strategy_type_time:
187  timeout = make_timeout_time_us(opt->timeout);
188  ok = scale_get_values_timeout(
189  sc,
190  &arr,
191  &len,
192  &timeout);
193  break;
194 
195  case strategy_type_samples:
196  default:
197  len = opt->samples;
198  ok = scale_get_values_samples(
199  sc,
200  &arr,
201  len);
202  break;
203  }
204 
205  //if an error occurred or no samples obtained
206  //deallocate any memory and then return false
207  if(!ok || len == 0) {
208  free(arr);
209  return false;
210  }
211 
212  switch(opt->read) {
213  case read_type_average:
214  util_average(arr, len, val);
215  break;
216 
217  case read_type_median:
218  default:
219  util_median(arr, len, val);
220  break;
221  }
222 
223  //at this point, statistical func is done
224  //so deallocate and return true
225  free(arr);
226  return true;
227 
228 }

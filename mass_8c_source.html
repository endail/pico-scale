26 int mass_to_string(
27  const mass_t* const m,
28  char* const buff) {
29 
30  assert(m != NULL);
31  assert(buff != NULL);
32 
33  double n; //value
34  mass_get_value(m, &n);
35  double i; //int part; discard
36  const double f = fabs(modf(n, &i)); //frac part
37  uint d = 0; //decimal count
38 
39  //if less than the epsilon, then it's ~0
40  if(f >= DBL_EPSILON) {
41  d = (uint)fmax(0, ceil(1 - log10(f)));
42  }
43 
44  return snprintf(
45  buff,
46  MASS_TO_STRING_BUFF_SIZE,
47  "%01.*f %s", //format
48  d, //how many decimals
49  n, //the value
50  mass_unit_to_string(m->unit)); //suffix (ie. "kg")
51 
52 }
